🌱 CarbonX API Client - Interactive Examples

This page demonstrates how to use the CarbonXAPI JavaScript client to interact with the backend.

📦 Installation

Include the API client in your HTML:

<script src="js/carbonx-api-client.js"></script>

🚀 Initialization

// Initialize the API client
const api = new CarbonXAPI('http://localhost:8080');

// Now you can use api.users.* methods

👥 Get All Users

// Get all users
const users = await api.users.getAll();

// Get users by role
const admins = await api.users.getAll({ role: 'admin' });

// Get active users
const activeUsers = await api.users.getAll({ active: true });

🔍 Find Specific User

// Get user by ID
const user = await api.users.getById('12345');

// Get user by email
const user = await api.users.getByEmail('john@example.com');

// Get user by username
const user = await api.users.getByUsername('johndoe');

➕ Create User

const newUser = await api.users.create({
    username: 'johndoe',
    email: 'john@example.com',
    password: 'securepass123',
    firstName: 'John',
    lastName: 'Doe',
    role: 'user',
    companyName: 'Acme Corp'
});

✏️ Update User

const updated = await api.users.update('12345', {
    firstName: 'Jonathan',
    lastName: 'Smith',
    role: 'admin'
});

🔐 User Login

try {
    const response = await api.users.login('john@example.com', 'password123');
    console.log('Login successful:', response.user);
    // Store user session, redirect, etc.
} catch (error) {
    console.error('Login failed:', error.message);
}

🔄 Activate/Deactivate User

// Activate user
await api.users.activate('12345');

// Deactivate user
await api.users.deactivate('12345');

🗑️ Delete User

await api.users.delete('12345');

⚠️ Error Handling

try {
    const user = await api.users.getById('invalid-id');
} catch (error) {
    console.error('Status:', error.status);
    console.error('Message:', error.message);
    
    if (error.status === 404) {
        alert('User not found');
    } else if (error.status === 0) {
        alert('Cannot connect to server');
    }
}

🔄 Real-World Example: Complete User Flow

// 1. Create a new user
const newUser = await api.users.create({
    username: 'janedoe',
    email: 'jane@example.com',
    password: 'pass123',
    firstName: 'Jane',
    lastName: 'Doe'
});

console.log('User created with ID:', newUser.user.id);

// 2. Login the user
const loginResponse = await api.users.login('jane@example.com', 'pass123');
console.log('Logged in as:', loginResponse.user.username);

// 3. Update user profile
await api.users.update(loginResponse.user.id, {
    companyName: 'TechCorp',
    role: 'supplier'
});

// 4. Get all users to verify
const allUsers = await api.users.getAll();
console.log('Total users:', allUsers.length);
